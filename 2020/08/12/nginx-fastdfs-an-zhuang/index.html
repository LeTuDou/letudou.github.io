Nginx + FastDfs安装
Nginx FastDfs
发布日期:   2020-08-12
文章字数:   725
阅读时长:   3 分
阅读次数:  

程序版本

Nginx V1.16.1

libfastcomman V1.0.43

fastdfs V6.06

fastdfs-nginx-module V1.22

依赖安装

yum -y install gcc make zlib zlib-devel gcc-c++ libtool  openssl openssl-devel

FastDfs目录创建

mkdir  -p /fastdfs/tracker
mkdir  -p /fastdfs/storage
mkdir  -p /fastdfs/logs
mkdir  -p /fastdfs/tmp

下载libfastcomman 源码安装（基础环境）

#   github address:  https://github.com/happyfish100/libfastcommon.git
#   gitee address:   https://gitee.com/fastdfs100/libfastcommon.git

 git clone https://github.com/happyfish100/libfastcommon.git
 cd libfastcommon
 ./make.sh clean && ./make.sh && ./make.sh install
 # 复制文件，解决FastDFS中lib配置文件路径问题。
cp /usr/lib64/libfastcommon.so /usr/local/lib/libfastcommon.so
cp /usr/lib64/libfastcommon.so /usr/lib/libfastcommon.so

下载安装 FastDFS 源码安装

# 安装完后 默认配置文件的目录为 /etc/fdfs/ 默认命令存放在 /usr/bin/ 并且以 fdfs_ 开头。

#   github address:  https://github.com/happyfish100/fastdfs.git
#   gitee address:   https://gitee.com/fastdfs100/fastdfs.git

git clone https://github.com/happyfish100/fastdfs.git
cd fastdfs
./make.sh clean
./make.sh 
./make.sh install
./setup.sh /etc/fdfs

修改配置文件 并启动

# 配置 tracker
vim tracker.conf
base_path =/fastdfs/tracker
# 启动 tracker
fdfs_trackerd /etc/fdfs/tracker.conf start

# 配置 storage
vim storage.conf
# 日志目录
base_path=/fastdfs/storage 
# 存储目录
store_path0=/fastdfs/storage
#tracker节点
tracker_server=192.168.174.133:22122
#启动storage
fdfs_storaged /etc/fdfs/storage.conf start

# 配置 client
vim client.conf
#tracker节点
tracker_server=192.168.174.133:22122
#日志路径
base_path=/fastdfs/logs

注:

# 若是 Linux 系统可以通过服务的方式操作。
/sbin/service fdfs_trackerd restart
/sbin/service fdfs_storaged restart
#监控程序
/usr/bin/fdfs_monitor /etc/fdfs/client.conf
#运行测试程序 /usr/bin/fdfs_test  
/usr/bin/fdfs_test /etc/fdfs/client.conf upload /etc/fdfs/client.conf.sample

Nginx 安装 与 配置fastdfs-nginx-module

# Nginx源码下载
wget http://nginx.org/download/nginx-1.16.1.tar.gz
tar -zxvf nginx-1.16.1.tar.gz
# fastdfs-nginx-module 源码下载
# github address: https://github.com/happyfish100/fastdfs-nginx-module
# gitee address: https://gitee.com/fastdfs100/fastdfs-nginx-module.git
git clone https://gitee.com/fastdfs100/fastdfs-nginx-module.git


# 配置mod_fastdfs.conf文件
cd /root/fastdfs/fastdfs-nginx-module/src
cp mod_fastdfs.conf /etc/fdfs/
vim /etc/fdfs/mod_fastdfs.conf
# 客户端访问文件连接超时时长（单位：秒）
connect_timeout=10
# 存储日志路径
base_path=/fastdfs/tmp
# tracker服务IP和端口
tracker_server=192.168.174.133:22122
# 访问链接前缀加上组名
url_have_group_name=true
group_count = 1
[group1]
# 和storage的groupname一一对应
group_name=group1
storage_server_port=23000
store_path_count=1
# 文件存储路径
store_path0=/fastdfs/storage

# 编译安装 PCRE (PCRE 作用是让 Nginx 支持 Rewrite 功能)
#wget http://downloads.sourceforge.net/project/pcre/pcre/8.35/pcre-8.35.tar.gz
#tar -zxvf pcre-8.35.tar.gz
#cd pcre-8.35
#./configure
#make && make install
#pcre-config --version

# 编译安装 nginx
cd nginx-1.16.1

#./configure  --prefix=/usr/local/webserver/nginx --with-http_stub_status_module --with-http_ssl_module --with-pcre=/root/pcre-8.35 --add-module=/root/fastdfs-nginx-module/src

./configure  --prefix=/usr/local/webserver/nginx --with-http_stub_status_module --with-http_ssl_module --add-module=/root/fastdfs-nginx-module/src
make
make install

# 配置nginx
vim /usr/local/webserver/nginx/conf/nginx.conf
# 监听域名中带有group0 到 group9 的，交给fastdfs-nginx-module模块处理
location ~/group([0-9])/ {
    ngx_fastdfs_module;
}
# 重启 nginx 使配置生效
/usr/local/webserver/nginx/sbin/nginx -s reload
# 测试
#/usr/bin/fdfs_test /etc/fdfs/client.conf upload /root/1.jpg

fdfs_upload_file /etc/fdfs/client.conf /root/1.jpg

# 访问测试
